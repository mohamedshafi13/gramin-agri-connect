📝 Complete SQL Schema
-- ═══════════════════════════════════════════════ -- AgriMarket Database Schema -- Compatible with MySQL 8.0+ / PostgreSQL 14+ -- ═══════════════════════════════════════════════ CREATE DATABASE agrimarket CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci; USE agrimarket; -- ── MODULE 1: USER MANAGEMENT ── CREATE TABLE users ( user_id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(100) NOT NULL, email VARCHAR(150) UNIQUE NOT NULL, phone VARCHAR(15) UNIQUE NOT NULL, password_hash VARCHAR(255) NOT NULL, role ENUM('farmer','buyer','admin') NOT NULL, avatar_url VARCHAR(500), location VARCHAR(200), is_verified BOOLEAN DEFAULT FALSE, is_active BOOLEAN DEFAULT TRUE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP ); CREATE TABLE farmer_profiles ( profile_id INT AUTO_INCREMENT PRIMARY KEY, user_id INT UNIQUE NOT NULL, farm_name VARCHAR(200), farm_size_acres DECIMAL(10,2), certification ENUM('none','organic','gai','fssai'), aadhar_no VARCHAR(12), bank_account VARCHAR(20), ifsc_code VARCHAR(11), rating DECIMAL(3,1) DEFAULT 0.0, total_sales DECIMAL(15,2) DEFAULT 0.00, FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE ); -- ── MODULE 2: MARKETPLACE ── CREATE TABLE categories ( category_id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(80) NOT NULL, icon VARCHAR(10), parent_id INT, FOREIGN KEY (parent_id) REFERENCES categories(category_id) ); CREATE TABLE products ( product_id INT AUTO_INCREMENT PRIMARY KEY, farmer_id INT NOT NULL, category_id INT, name VARCHAR(150) NOT NULL, description TEXT, price_per_unit DECIMAL(10,2) NOT NULL, unit ENUM('kg','quintal','unit','litre','dozen'), stock_qty DECIMAL(10,2) NOT NULL, min_order_qty DECIMAL(10,2) DEFAULT 1, is_organic BOOLEAN DEFAULT FALSE, harvest_date DATE, expiry_date DATE, images JSON, status ENUM('active','inactive','sold_out') DEFAULT 'active', created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (farmer_id) REFERENCES users(user_id), FOREIGN KEY (category_id) REFERENCES categories(category_id) ); CREATE TABLE product_reviews ( review_id INT AUTO_INCREMENT PRIMARY KEY, product_id INT NOT NULL, buyer_id INT NOT NULL, rating TINYINT CHECK (rating BETWEEN 1 AND 5), comment TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (product_id) REFERENCES products(product_id), FOREIGN KEY (buyer_id) REFERENCES users(user_id) ); -- ── MODULE 3: ORDERS & TRANSACTIONS ── CREATE TABLE orders ( order_id INT AUTO_INCREMENT PRIMARY KEY, order_code VARCHAR(20) UNIQUE GENERATED ALWAYS AS (CONCAT('AM-', order_id)) STORED, buyer_id INT NOT NULL, product_id INT NOT NULL, farmer_id INT NOT NULL, quantity DECIMAL(10,2) NOT NULL, unit_price DECIMAL(10,2) NOT NULL, total_amount DECIMAL(12,2) NOT NULL, platform_fee DECIMAL(10,2), delivery_charge DECIMAL(8,2) DEFAULT 0.00, status ENUM('pending','confirmed','processing','shipped','delivered','cancelled','refunded') DEFAULT 'pending', delivery_address JSON, expected_delivery DATE, notes TEXT, ordered_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (buyer_id) REFERENCES users(user_id), FOREIGN KEY (product_id) REFERENCES products(product_id), FOREIGN KEY (farmer_id) REFERENCES users(user_id) ); CREATE TABLE order_tracking ( tracking_id INT AUTO_INCREMENT PRIMARY KEY, order_id INT NOT NULL, status VARCHAR(80), description TEXT, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (order_id) REFERENCES orders(order_id) ); -- ── MODULE 4: PAYMENT ── CREATE TABLE payments ( payment_id INT AUTO_INCREMENT PRIMARY KEY, order_id INT NOT NULL, payer_id INT NOT NULL, amount DECIMAL(12,2) NOT NULL, method ENUM('upi','card','netbanking','wallet','cod'), gateway VARCHAR(50), txn_id VARCHAR(100) UNIQUE, gateway_ref VARCHAR(200), status ENUM('initiated','success','failed','refunded','pending'), paid_at TIMESTAMP, FOREIGN KEY (order_id) REFERENCES orders(order_id), FOREIGN KEY (payer_id) REFERENCES users(user_id) ); CREATE TABLE farmer_payouts ( payout_id INT AUTO_INCREMENT PRIMARY KEY, farmer_id INT NOT NULL, order_id INT NOT NULL, gross_amount DECIMAL(12,2), platform_fee DECIMAL(10,2), net_amount DECIMAL(12,2), utr_number VARCHAR(50), status ENUM('pending','processed','failed'), paid_at TIMESTAMP, FOREIGN KEY (farmer_id) REFERENCES users(user_id), FOREIGN KEY (order_id) REFERENCES orders(order_id) ); -- ── MODULE 5: MESSAGING ── CREATE TABLE conversations ( convo_id INT AUTO_INCREMENT PRIMARY KEY, user_one INT NOT NULL, user_two INT NOT NULL, order_id INT, last_msg_at TIMESTAMP, UNIQUE KEY (user_one, user_two), FOREIGN KEY (user_one) REFERENCES users(user_id), FOREIGN KEY (user_two) REFERENCES users(user_id) ); CREATE TABLE messages ( message_id INT AUTO_INCREMENT PRIMARY KEY, convo_id INT NOT NULL, sender_id INT NOT NULL, content TEXT NOT NULL, msg_type ENUM('text','image','file','offer') DEFAULT 'text', is_read BOOLEAN DEFAULT FALSE, sent_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (convo_id) REFERENCES conversations(convo_id), FOREIGN KEY (sender_id) REFERENCES users(user_id) ); -- ── MODULE 6: KNOWLEDGE HUB ── CREATE TABLE knowledge_articles ( article_id INT AUTO_INCREMENT PRIMARY KEY, title VARCHAR(255) NOT NULL, category ENUM('soil','pest','water','organic','market','weather','scheme'), content LONGTEXT, summary TEXT, author VARCHAR(100), tags JSON, media_urls JSON, views INT DEFAULT 0, is_published BOOLEAN DEFAULT TRUE, published_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- ── USEFUL VIEWS ── CREATE VIEW vw_order_details AS SELECT o.order_id, o.order_code, o.status, o.total_amount, b.name AS buyer_name, f.name AS farmer_name, p.name AS product_name, o.quantity, o.ordered_at FROM orders o JOIN users b ON o.buyer_id = b.user_id JOIN users f ON o.farmer_id = f.user_id JOIN products p ON o.product_id = p.product_id; CREATE VIEW vw_farmer_earnings AS SELECT fp.farmer_id, u.name, SUM(fp.net_amount) AS total_earned, COUNT(fp.payout_id) AS total_payouts FROM farmer_payouts fp JOIN users u ON fp.farmer_id = u.user_id GROUP BY fp.farmer_id, u.name; -- ── INDEXES FOR PERFORMANCE ── CREATE INDEX idx_products_farmer ON products(farmer_id); CREATE INDEX idx_orders_buyer ON orders(buyer_id); CREATE INDEX idx_orders_status ON orders(status); CREATE INDEX idx_messages_convo ON messages(convo_id, sent_at); CREATE FULLTEXT INDEX idx_products_search ON products(name, description); CREATE FULLTEXT INDEX idx_articles_search ON knowledge_articles(title, content);

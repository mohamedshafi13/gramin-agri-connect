-- ═══════════════════════════════════════════════ -- AgriMarket Buyer Portal — Database Schema -- Compatible with MySQL 8.0+ / PostgreSQL 14+ -- ═══════════════════════════════════════════════ CREATE DATABASE agrimarket_buyer CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci; USE agrimarket_buyer; -- ── MODULE 1: BUYERS ── CREATE TABLE users ( user_id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(100) NOT NULL, email VARCHAR(150) UNIQUE NOT NULL, phone VARCHAR(15) UNIQUE NOT NULL, password_hash VARCHAR(255) NOT NULL, role ENUM('buyer') DEFAULT 'buyer', avatar_url VARCHAR(500), delivery_address JSON, is_verified BOOLEAN DEFAULT FALSE, is_active BOOLEAN DEFAULT TRUE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- ── MODULE 2: MARKETPLACE (READ) ── CREATE TABLE products ( product_id INT AUTO_INCREMENT PRIMARY KEY, farmer_id INT NOT NULL, name VARCHAR(150) NOT NULL, category ENUM('grains','vegetables','fruits','spices'), price_per_unit DECIMAL(10,2) NOT NULL, unit ENUM('kg','quintal','unit','dozen'), stock_qty DECIMAL(10,2) NOT NULL, min_order_qty DECIMAL(10,2) DEFAULT 1, is_organic BOOLEAN DEFAULT FALSE, status ENUM('active','sold_out') DEFAULT 'active', FOREIGN KEY (farmer_id) REFERENCES users(user_id) ); -- ── MODULE 3: ORDERS & TRANSACTIONS ── CREATE TABLE orders ( order_id INT AUTO_INCREMENT PRIMARY KEY, order_code VARCHAR(20) UNIQUE, buyer_id INT NOT NULL, product_id INT NOT NULL, farmer_id INT NOT NULL, quantity DECIMAL(10,2) NOT NULL, unit_price DECIMAL(10,2) NOT NULL, total_amount DECIMAL(12,2) NOT NULL, delivery_charge DECIMAL(8,2) DEFAULT 0.00, discount_amount DECIMAL(8,2) DEFAULT 0.00, promo_code VARCHAR(30), status ENUM('pending','confirmed','processing','shipped','delivered','cancelled','refunded') DEFAULT 'pending', delivery_address JSON, expected_delivery DATE, rated BOOLEAN DEFAULT FALSE, ordered_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (buyer_id) REFERENCES users(user_id), FOREIGN KEY (product_id) REFERENCES products(product_id) ); -- ── MODULE 4: PAYMENT ── CREATE TABLE payments ( payment_id INT AUTO_INCREMENT PRIMARY KEY, order_id INT NOT NULL, buyer_id INT NOT NULL, amount DECIMAL(12,2) NOT NULL, method ENUM('upi','card','netbanking','wallet','cod'), gateway VARCHAR(50), txn_id VARCHAR(100) UNIQUE, status ENUM('initiated','success','failed','refunded'), refund_amount DECIMAL(12,2), paid_at TIMESTAMP, FOREIGN KEY (order_id) REFERENCES orders(order_id), FOREIGN KEY (buyer_id) REFERENCES users(user_id) ); -- ── MODULE 5: MESSAGING ── CREATE TABLE conversations ( convo_id INT AUTO_INCREMENT PRIMARY KEY, buyer_id INT NOT NULL, farmer_id INT NOT NULL, order_id INT, last_msg_at TIMESTAMP, UNIQUE KEY (buyer_id, farmer_id), FOREIGN KEY (buyer_id) REFERENCES users(user_id), FOREIGN KEY (farmer_id) REFERENCES users(user_id) ); CREATE TABLE messages ( message_id INT AUTO_INCREMENT PRIMARY KEY, convo_id INT NOT NULL, sender_id INT NOT NULL, content TEXT NOT NULL, msg_type ENUM('text','image','offer') DEFAULT 'text', is_read BOOLEAN DEFAULT FALSE, sent_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (convo_id) REFERENCES conversations(convo_id), FOREIGN KEY (sender_id) REFERENCES users(user_id) ); -- ── MODULE 6: WISHLIST ── CREATE TABLE wishlist ( wishlist_id INT AUTO_INCREMENT PRIMARY KEY, buyer_id INT NOT NULL, product_id INT NOT NULL, notes TEXT, added_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, UNIQUE KEY (buyer_id, product_id), FOREIGN KEY (buyer_id) REFERENCES users(user_id), FOREIGN KEY (product_id) REFERENCES products(product_id) ); -- ── REVIEWS ── CREATE TABLE product_reviews ( review_id INT AUTO_INCREMENT PRIMARY KEY, product_id INT NOT NULL, order_id INT NOT NULL, buyer_id INT NOT NULL, rating TINYINT CHECK (rating BETWEEN 1 AND 5), comment TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (product_id) REFERENCES products(product_id), FOREIGN KEY (buyer_id) REFERENCES users(user_id) ); -- ── USEFUL VIEWS ── CREATE VIEW vw_buyer_orders AS SELECT o.order_code, o.status, o.total_amount, o.ordered_at, p.name AS product_name, u.name AS farmer_name, o.quantity, o.expected_delivery FROM orders o JOIN products p ON o.product_id = p.product_id JOIN users u ON o.farmer_id = u.user_id; -- ── INDEXES ── CREATE INDEX idx_orders_buyer ON orders(buyer_id); CREATE INDEX idx_orders_status ON orders(status); CREATE INDEX idx_payments_buyer ON payments(buyer_id); CREATE INDEX idx_messages_convo ON messages(convo_id, sent_at); CREATE FULLTEXT INDEX idx_products_search ON products(name);
